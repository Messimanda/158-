    Serial.print(F("to station + softap ok\r\n"));

  if (wifi.joinAP(SSID, PASSWORD)) {

    Serial.print(F("IP:"));

  } else {

  }



  if (wifi.disableMUX()) {

    Serial.print(F("single ok\r\n"));

  } else {

    Serial.print(F("single err\r\n"));

  }



  Serial.print(F("setup end\r\n"));
